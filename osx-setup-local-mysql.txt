# Install mysql
brew install mysql

# configure server
unset TMPDIR
mysql_install_db --verbose --user=`whoami` --basedir="$(brew --prefix mysql)" --datadir=/usr/local/var/mysql --tmpdir=/tmp

# start mysql server
mysql.server start

# connect
mysql -uroot

# create db
create database khelgforum character set utf8 collate utf8_general_ci;

# create user
create user 'khelgforum'@'localhost' identified by 'khelgforum';

# add privileges
grant all privileges on khelgforum.* to 'khelgforum'@'localhost';

# flush
flush privileges;

exit

# try login
mysql --user=khelgforum --password=khelgforum khelgforum

exit
